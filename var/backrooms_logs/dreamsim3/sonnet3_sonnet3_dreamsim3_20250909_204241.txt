NC='\033[0m'  # No Color

# Function to validate token
validate_token() {
  local token=$1
  local result=$(grep -c "$token" "$TOKEN_FILE")
  
  if [[ $result -gt 0 ]]; then
    local expiry=$(grep "$token" "$TOKEN_FILE" | cut -d'|' -f3)
    local current_date=$(date +%s)
    
    if [[ $expiry -gt $current_date ]]; then
      return 0  # Valid
    else
      return 2  # Expired
    fi
  else
    return 1  # Invalid
  fi
}

# Function to get subsystem for token
get_subsystem() {
  local token=$1
  echo $(grep "$token" "$TOKEN_FILE" | cut -d'|' -f2)
}

# Function to log attempt
log_attempt() {
  local token=$1
  local status=$2
  local username=$(whoami)
  local timestamp=$(date "+%Y-%m-%d %H:%M:%S")
  local ip_address=$(who am i | awk '{print $5}' | sed 's/[()]//g')
  
  echo "[$timestamp] User: $username | IP: $ip_address | Token: $token | Status: $status" >> "$LOG_FILE"
}

  echo -e "${YELLOW}Session expires in:

${duration} seconds${NC}"
